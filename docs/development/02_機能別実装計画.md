```csharp
        Free,

        Premium,
        PremiumBYOK,
        Enterprise
    }

    public bool CanUseSplit() => _currentPlan != LicensePlan.Free;
```

```csharp
// PageDragDropHandler.cs
public class PageDragDropHandler
{
    public void OnPageDragStart(PdfPageViewModel page)
    {
        var data = new DataObject();
        data.SetData("PdfPage", page);
        data.SetData("SourcePdf", page.ParentPdf);
        
        DragDrop.DoDragDrop(page.Visual, data, DragDropEffects.Move | DragDropEffects.Copy);
    }
    
    public void OnPageDrop(object sender, DragEventArgs e)
    {
        if (e.Data.GetDataPresent("PdfPage"))
        {
            var page = e.Data.GetData("PdfPage") as PdfPageViewModel;
            var targetPdf = (sender as FrameworkElement)?.DataContext as PdfViewModel;
            
            if (targetPdf != null && page != null)
            {
                // ページを挿入
                InsertPageIntoPdf(page, targetPdf, GetDropIndex(e));
            }
        }
    }
    
    private void InsertPageIntoPdf(PdfPageViewModel page, PdfViewModel targetPdf, int index)
    {
        // PdfSharpでページ挿入
        using var sourcePdf = PdfReader.Open(page.ParentPdf.FilePath, PdfDocumentOpenMode.Import);
        using var targetPdfDoc = PdfReader.Open(targetPdf.FilePath, PdfDocumentOpenMode.Modify);
        
        var pageToInsert = sourcePdf.Pages[page.PageIndex];
        targetPdfDoc.Pages.Insert(index, pageToInsert);
        
        targetPdfDoc.Save(targetPdf.FilePath);
        
        // UI更新
        targetPdf.ReloadPages();
    }
}
```

```csharp
// OfficeConverter.cs
public class OfficeConverter
{
    public async Task<string> ConvertWordToPdf(string wordPath, string outputPath)
    {
        return await Task.Run(() =>
        {
            var wordApp = new Microsoft.Office.Interop.Word.Application();
            try
            {
                var doc = wordApp.Documents.Open(wordPath);
                doc.SaveAs2(outputPath, 
                    Microsoft.Office.Interop.Word.WdSaveFormat.wdFormatPDF);
                doc.Close();
                
                return outputPath;
            }
            finally
            {
                wordApp.Quit();
            }
        });
    }
    
    public async Task<string> ConvertExcelToPdf(string excelPath, string outputPath)
    {
        return await Task.Run(() =>
        {
            var excelApp = new Microsoft.Office.Interop.Excel.Application();
            try
            {
                var workbook = excelApp.Workbooks.Open(excelPath);
                workbook.ExportAsFixedFormat(
                    Microsoft.Office.Interop.Excel.XlFixedFormatType.xlTypePDF,
                    outputPath);
                workbook.Close();
                
                return outputPath;
            }
            finally
            {
                excelApp.Quit();
            }
        });
    }
}
```

```csharp
// AISummaryService.cs
public class AISummaryService
{
    private readonly HttpClient _httpClient;
    private readonly string _apiKey;
    
    public async Task<string> SummarizePdf(string pdfPath, SummaryLevel level)
    {
        // PDFからテキスト抽出
        var text = ExtractTextFromPdf(pdfPath);
        
        if (text.Length > 50000)
        {
            // 長すぎる場合は分割して要約
            return await SummarizeLongDocument(text, level);
        }
        
        // Claude APIリクエスト
        var prompt = level switch
        {
            SummaryLevel.Brief => "以下の文書を3-5文で簡潔に要約してください。",
            SummaryLevel.Standard => "以下の文書を1段落（200-300文字）で要約してください。",
            SummaryLevel.Detailed => "以下の文書を複数段落で詳しく要約してください。主要なポイントをすべて含めてください。",
            _ => "以下の文書を要約してください。"
        };
        
        var request = new
        {
            model = "claude-sonnet-4-20250514",
            max_tokens = level == SummaryLevel.Detailed ? 2000 : 1000,
            messages = new[]
            {
                new { role = "user", content = $"{prompt}\n\n{text}" }
            }
        };
        
        var response = await _httpClient.PostAsJsonAsync(
            "https://api.anthropic.com/v1/messages",
```

```csharp
// SemanticSearchService.cs
public class SemanticSearchService
{
    private readonly HttpClient _httpClient;
    private readonly string _openAiApiKey;
    private readonly QdrantClient _vectorDb;
    
    public async Task IndexDocument(string pdfPath)
    {
        // PDFからテキスト抽出
        var text = ExtractTextFromPdf(pdfPath);
        
        // チャンクに分割（1000文字ずつ）
        var chunks = SplitIntoChunks(text, 1000);
        
        foreach (var (chunk, index) in chunks.Select((c, i) => (c, i)))
        {
            // OpenAI Embeddings APIでベクトル化
            var embedding = await GetEmbedding(chunk);
            
            // Qdrantに保存
            await _vectorDb.UpsertAsync(new
            {
                id = $"{pdfPath}_{index}",
                vector = embedding,
                payload = new
                {
                    pdf_path = pdfPath,
                    chunk_index = index,
                    text = chunk
                }
            });
        }
    }
    
    public async Task<List<SearchResult>> Search(string query, int topK = 10)
    {
        // クエリをベクトル化
        var queryEmbedding = await GetEmbedding(query);
        
        // Qdrantで類似度検索
        var results = await _vectorDb.SearchAsync(queryEmbedding, topK);
        
        return results.Select(r => new SearchResult
        {
            PdfPath = r.Payload["pdf_path"],
            ChunkIndex = r.Payload["chunk_index"],
            Text = r.Payload["text"],
            Score = r.Score
        }).ToList();
    }
    
    private async Task<float[]> GetEmbedding(string text)
    {
        var request = new
        {
            model = "text-embedding-3-small",
            input = text
        };
        
        var response = await _httpClient.PostAsJsonAsync(
            "https://api.openai.com/v1/embeddings",
            request
        );
        
        var result = await response.Content.ReadFromJsonAsync<EmbeddingResponse>();
        return result.Data[0].Embedding;
    }
}
```

```csharp
// CommandConsole.cs
public class CommandConsole
{
    private readonly Dictionary<string, ICommand> _commands;
    
    public CommandConsole()
    {
        _commands = new Dictionary<string, ICommand>
        {
            ["merge-pdf"] = new MergePdfCommand(),
            ["split-pdf"] = new SplitPdfCommand(),
            ["rename-pdf"] = new RenamePdfCommand(),
            ["summarize-pdf"] = new SummarizePdfCommand(),
            ["search-pdf"] = new SearchPdfCommand(),
        };
    }
    
    public async Task<CommandResult> ExecuteAsync(string commandLine)
    {
        var (commandName, args) = ParseCommandLine(commandLine);
        
        if (!_commands.TryGetValue(commandName, out var command))
        {
            return CommandResult.Error($"Unknown command: {commandName}");
        }
        
        try
        {
            return await command.ExecuteAsync(args);
        }
        catch (Exception ex)
        {
            return CommandResult.Error(ex.Message);
        }
    }
}

public interface ICommand
{
    Task<CommandResult> ExecuteAsync(Dictionary<string, object> args);
}
```
